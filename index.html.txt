  <!-- SCRIPTS: simple WebAudio pad for calm simulation + mobile nav -->
  <script>
    // mobile simple nav toggle (if needed later)

    // WebAudio gentle pad + pulsing gain to simulate massage rhythm
    let audioCtx = null;
    let osc = null;
    let gain = null;
    let lfo = null;

    const playBtn = document.getElementById('play');
    const stopBtn = document.getElementById('stop');
    const status = document.getElementById('calm-status');

    function startPad(){
      if(!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
      osc = audioCtx.createOscillator();
      gain = audioCtx.createGain();
      lfo = audioCtx.createOscillator();
      let lfoGain = audioCtx.createGain();

      osc.type = 'sine';
      osc.frequency.value = 180; // soft high-ish pad

      gain.gain.value = 0.0001; // start almost silent

      // LFO to pulse the gain slowly (massage rhythm)
      lfo.type = 'sine';
      lfo.frequency.value = 0.5; // 0.5 Hz -> 2s per cycle
      lfoGain.gain.value = 0.06;

      lfo.connect(lfoGain);
      lfoGain.connect(gain.gain);

      osc.connect(gain);
      gain.connect(audioCtx.destination);

      osc.start();
      lfo.start();

      // ramp up gently
      gain.gain.exponentialRampToValueAtTime(0.02, audioCtx.currentTime + 1.2);
      status.textContent = '播放中 • 輕柔脈動';
    }

    function stopPad(){
      if(!audioCtx) return;
      try{
        gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 0.8);
        setTimeout(()=>{
          osc.stop();
          lfo.stop();
          osc.disconnect();
          lfo.disconnect();
          gain.disconnect();
          audioCtx.close();
          audioCtx = null;
          status.textContent = '已停止';
        },1000);
      }catch(e){
        status.textContent = '已停止';
      }
    }

    playBtn?.addEventListener('click', ()=>{
      // resume context if suspended
      if(audioCtx && audioCtx.state === 'suspended') audioCtx.resume();
      if(!audioCtx) startPad();
    });
    stopBtn?.addEventListener('click', stopPad);
  </script>
